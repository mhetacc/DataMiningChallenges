\subsection{Lasso Regression Progressive Filtering And Transformations Results}
\label{app:phone_lasso_filtering}

\begin{itemize}
        \item No filtering: $MSE = 18091358$;
        \item Only log-transform target:$MSE = 6.200236$;
        \item Filtered incoming calls, SMS and calls to the call center: $MSE = 18040253$;
        \item Filtered incoming calls, SMS: $MSE = 18080968$;
        \item Only Value of calls filtered out: $MSE = 18067089$;
        \item Only amount of calls filtered out: $MSE = 18118489$;
        \item Incoming calls, SMS, call-center calls, and payment information filtered out: $MSE = 18035221$;
        \item Incoming calls, SMS, call-center calls, payment information, and sex filtered out: $MSE = 18036839$;
        \item Incoming calls, SMS, call-center calls, payment information, sex, activation zone, and activation channel filtered out: $MSE = 18034269$;
        \item Incoming calls, SMS, call-center calls, payment information, sex, activation zone, activation channel, and age filtered out: $MSE = 18034610$;
        \item Incoming calls, SMS, call-center calls, payment information, sex, activation zone, activation channel, age, and plan information filtered out: $MSE = 18975290$;
        \item Incoming calls, SMS, call-center calls, payment information, sex, activation zone, activation channel, age, plan information, and value-added services filtered out: $MSE = 18969265$;
        \item Incoming calls, SMS, call-center calls, payment information, value-added services, activation zone, and activation channel filtered out: $MSE = 18021270$;
        \item Incoming calls, SMS, call-center calls, payment information, value-added services, and activation zone filtered out: $MSE = 18022544$;
        \item Incoming calls, SMS, call-center calls, payment information, value-added services, and activation channel filtered out: $MSE = 18024845$;
        \item Incoming calls, SMS, call-center calls, payment information, value-added service 1, activation zone, and activation channel filtered out: $MSE = 18035228$;
        \item Incoming calls, SMS, call-center calls, payment information, value-added service 2, activation zone, and activation channel filtered out: $\mathbf{MSE = 18019454}$ \textbf{(best)}.
\end{itemize}

Transformations:
\begin{itemize}
        \item Peak and off-peak values aggregated together: $MSE = 32024591$;
        \item Peak and off-peak values aggregated together, with prediction performed on average call time, number of calls, and value over nine months: $MSE = 32174039$.
\end{itemize}

With best filtering:
\begin{itemize}
        \item Log-transform of the target variable only: $MSE = 6.253648$;
        \item Log-transform of the monthly predictors only: $MSE = 20415157$;
        \item Log-transform of both the target variable and the monthly predictors: $MSE = 4.289544$.
\end{itemize}
